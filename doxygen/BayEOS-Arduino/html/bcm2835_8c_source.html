72 // Get raspberry PI model version
73 int bcm2835_get_pi_version( void )
74 {
75  int rev = 0;
76  char buff[512];
77  char * p;
78  char * pend;
79 
80  FILE * fd ;
81 
82  // do some clean up
83  memset(buff,0,sizeof(buff));
84 
85  fd = fopen("/proc/cpuinfo","r");
86 
87  // Opened successfully
88  if( fd )
89  {
90  //printf("File opened successfully through fopen()\n");
91 
92  // parse each line until we the end or we find the good one
93  while( fgets(buff, sizeof(buff), fd) != NULL && rev ==0 )
94  {
95  // search
96  if( (strstr(buff, "Revision" )) != NULL )
97  {
98  // point to the separator ":" format is has follow
99  // Revision : 000f
100  if ( (p = strtok( buff, ":")) != NULL )
101  {
102  // Ok get value
103  if ( (p = strtok( NULL, ":")) != NULL )
104  {
105  // Revision Version is in hex format so put 0x before the number
106  *p = 'x';
107  *--p = '0';
108 
109  // convert to number
110  rev = strtol(p, &pend, 16);
111 
112  //printf("rev=%d 0x%04x\n", rev, rev);
113 
114  // not Okay ?
115  if ( !*pend )
116  {
117  rev= 0;
118  }
119  else
120  {
121  // Revision 1 or 2 ?
122  rev = (rev < 4 ) ? 1 : 2 ;
123  }
124  }
125  }
126  }
127  }
128 
129  // Close the file.
130  if(fd)
131  {
132  fclose(fd);
133  }
134  }
135 
136  return rev;
137 }

218 // Function select
219 // pin is a BCM2835 GPIO pin number NOT RPi pin number
220 // There are 6 control registers, each control the functions of a block
221 // of 10 pins.
222 // Each control register has 10 sets of 3 bits per GPIO pin:
223 //
224 // 000 = GPIO Pin X is an input
225 // 001 = GPIO Pin X is an output
226 // 100 = GPIO Pin X takes alternate function 0
227 // 101 = GPIO Pin X takes alternate function 1
228 // 110 = GPIO Pin X takes alternate function 2
229 // 111 = GPIO Pin X takes alternate function 3
230 // 011 = GPIO Pin X takes alternate function 4
231 // 010 = GPIO Pin X takes alternate function 5
232 //
233 // So the 3 bits for port X are:
234 // X / 10 + ((X % 10) * 3)
235 void bcm2835_gpio_fsel(uint8_t pin, uint8_t mode)
236 {
237  // Function selects are 10 pins per 32 bit word, 3 bits per pin
238  volatile uint32_t* paddr = bcm2835_gpio + BCM2835_GPFSEL0/4 + (pin/10);
239  uint8_t shift = (pin % 10) * 3;
240  uint32_t mask = BCM2835_GPIO_FSEL_MASK << shift;
241  uint32_t value = mode << shift;
242  bcm2835_peri_set_bits(paddr, value, mask);
243 }

442 // microseconds
443 void bcm2835_delayMicroseconds(uint64_t micros)
444 {
445  struct timespec t1;
446  uint64_t start;
447 
448  // Calling nanosleep() takes at least 100-200 us, so use it for
449  // long waits and use a busy wait on the System Timer for the rest.
450  start = bcm2835_st_read();
451 
452  if (micros > 450)
453  {
454  t1.tv_sec = 0;
455  t1.tv_nsec = 1000 * (long)(micros - 200);
456  nanosleep(&t1, NULL);
457  }
458 
459  bcm2835_st_delay(start, micros);
460 }

504 // Set the pullup/down resistor for a pin
505 //
506 // The GPIO Pull-up/down Clock Registers control the actuation of internal pull-downs on
507 // the respective GPIO pins. These registers must be used in conjunction with the GPPUD
508 // register to effect GPIO Pull-up/down changes. The following sequence of events is
509 // required:
510 // 1. Write to GPPUD to set the required control signal (i.e. Pull-up or Pull-Down or neither
511 // to remove the current Pull-up/down)
512 // 2. Wait 150 cycles ? this provides the required set-up time for the control signal
513 // 3. Write to GPPUDCLK0/1 to clock the control signal into the GPIO pads you wish to
514 // modify ? NOTE only the pads which receive a clock will be modified, all others will
515 // retain their previous state.
516 // 4. Wait 150 cycles ? this provides the required hold time for the control signal
517 // 5. Write to GPPUD to remove the control signal
518 // 6. Write to GPPUDCLK0/1 to remove the clock
519 //
520 // RPi has P1-03 and P1-05 with 1k8 pullup resistor
521 void bcm2835_gpio_set_pud(uint8_t pin, uint8_t pud)
522 {
523  bcm2835_gpio_pud(pud);
524  delayMicroseconds(10);
525  bcm2835_gpio_pudclk(pin, 1);
526  delayMicroseconds(10);
527  bcm2835_gpio_pud(BCM2835_GPIO_PUD_OFF);
528  bcm2835_gpio_pudclk(pin, 0);
529 }

531 void bcm2835_spi_begin(uint8_t cs)
532 {
533  volatile uint32_t* paddr = bcm2835_spi0 + BCM2835_SPI0_CS/4;
534 
535  // Set the SPI0 pins to the Alt 0 function to enable SPI0 access on them
536  // except if we need custom Chip Select Pin
537  // printf("bcm2835_spi_begin -> spi_custom_cs = %d \n",cs );
538 
539  // Do we need custom chip select control or
540  // drive CE1 manually (because CE1 does not work with hardware)
541  if ( cs > BCM2835_SPI_CS_NONE || cs == BCM2835_SPI_CS1 )
542  {
543  // indicate we will use a custom GPIO port
544  spi_custom_cs = cs ;
545 
546  // ok hard CE1 not working, drive it manually
547  if (cs == BCM2835_SPI_CS1 )
548  {
549  // Dirty Hack CE1 in now custom Chip Select GPIO 26
550  // the real CE1 pin
551  spi_custom_cs = RPI_GPIO_P1_26 ;
552 
553  bcm2835_gpio_fsel(spi_custom_cs, BCM2835_GPIO_FSEL_OUTP);
554  bcm2835_gpio_write(spi_custom_cs, HIGH);
555  }
556 
557  // Mask in we use custom CS (not sure it has a real effect)
558  bcm2835_peri_set_bits(paddr, BCM2835_SPI_CS_NONE, BCM2835_SPI0_CS_CS);
559  }
560  // Ok hardware driving of chip select
561  else
562  {
563  // Just in case
564  spi_custom_cs = 0 ;
565 
566  // Mask in the CS bits of CS
567  bcm2835_peri_set_bits(paddr, cs, BCM2835_SPI0_CS_CS);
568  }
569 
570  // Now we can drive the I/O as asked
571  if (spi_custom_cs == 0)
572  {
573  // Not custom CS, so hardware driven
574  bcm2835_gpio_fsel(RPI_GPIO_P1_24, BCM2835_GPIO_FSEL_ALT0); // CE0
575  bcm2835_gpio_fsel(RPI_GPIO_P1_26, BCM2835_GPIO_FSEL_ALT0); // CE1
576  }
577  else
578  {
579  // so set custom CS as output, High level by default
580  bcm2835_gpio_fsel(spi_custom_cs, BCM2835_GPIO_FSEL_OUTP); // Custom GPIO
581  bcm2835_gpio_write(spi_custom_cs, HIGH);
582  }
583 
584  // Classic pin, hardware driven
585  bcm2835_gpio_fsel(RPI_GPIO_P1_21, BCM2835_GPIO_FSEL_ALT0); // MISO
586  bcm2835_gpio_fsel(RPI_GPIO_P1_19, BCM2835_GPIO_FSEL_ALT0); // MOSI
587  bcm2835_gpio_fsel(RPI_GPIO_P1_23, BCM2835_GPIO_FSEL_ALT0); // CLK
588 
589  // Set the SPI CS register to the some sensible defaults
590  bcm2835_peri_write(paddr, 0); // All 0s
591 
592  // Clear TX and RX fifos
593  bcm2835_peri_write_nb(paddr, BCM2835_SPI0_CS_CLEAR);
594 }

596 void bcm2835_spi_end(void)
597 {
598  // Set all the SPI0 pins back to input
599  if (spi_custom_cs == 0)
600  {
601  bcm2835_gpio_fsel(RPI_GPIO_P1_26, BCM2835_GPIO_FSEL_INPT); // CE1
602  bcm2835_gpio_fsel(RPI_GPIO_P1_24, BCM2835_GPIO_FSEL_INPT); // CE0
603  }
604  else
605  {

850 // Writes an number of bytes to I2C
851 uint8_t bcm2835_i2c_write(const char * buf, uint32_t len)
852 {
853  volatile uint32_t* dlen = bcm2835_bsc1 + BCM2835_BSC_DLEN/4;
854  volatile uint32_t* fifo = bcm2835_bsc1 + BCM2835_BSC_FIFO/4;
855  volatile uint32_t* status = bcm2835_bsc1 + BCM2835_BSC_S/4;
856  volatile uint32_t* control = bcm2835_bsc1 + BCM2835_BSC_C/4;
857 
858  uint32_t remaining = len;
859  uint32_t i = 0;
860  uint8_t reason = BCM2835_I2C_REASON_OK;
861 
862  // Clear FIFO
863  bcm2835_peri_set_bits(control, BCM2835_BSC_C_CLEAR_1 , BCM2835_BSC_C_CLEAR_1 );
864  // Clear Status
865  bcm2835_peri_write_nb(status, BCM2835_BSC_S_CLKT | BCM2835_BSC_S_ERR | BCM2835_BSC_S_DONE);
866  // Set Data Length
867  bcm2835_peri_write_nb(dlen, len);
868  // Enable device and start transfer
869  bcm2835_peri_write_nb(control, BCM2835_BSC_C_I2CEN | BCM2835_BSC_C_ST);
870 
871  while (!(bcm2835_peri_read(status) & BCM2835_BSC_S_DONE))
872  {
873  while ((bcm2835_peri_read(status) & BCM2835_BSC_S_TXD) && remaining)
874  {
875  // Write to FIFO, no barrier
876  bcm2835_peri_write_nb(fifo, buf[i]);
877  i++;
878  remaining--;
879  }
880  // When remaining data is to be send, then wait for an empty FIFO
881  if (remaining >= BCM2835_BSC_FIFO_SIZE)
882  delayMicroseconds(i2c_byte_wait_us * BCM2835_BSC_FIFO_SIZE);
883  else
884  delayMicroseconds(i2c_byte_wait_us * remaining);
885  }
886 
887  // Received a NACK
888  if (bcm2835_peri_read(status) & BCM2835_BSC_S_ERR)
889  {
890  reason = BCM2835_I2C_REASON_ERROR_NACK;
891  }
892 
893  // Received Clock Stretch Timeout
894  else if (bcm2835_peri_read(status) & BCM2835_BSC_S_CLKT)
895  {
896  reason = BCM2835_I2C_REASON_ERROR_CLKT;
897  }
898 
899  // Not all data is sent
900  else if (remaining)
901  {
902  reason = BCM2835_I2C_REASON_ERROR_DATA;
903  }
904 
905  bcm2835_peri_set_bits(control, BCM2835_BSC_S_DONE , BCM2835_BSC_S_DONE);
906 
907  return reason;
908 }
