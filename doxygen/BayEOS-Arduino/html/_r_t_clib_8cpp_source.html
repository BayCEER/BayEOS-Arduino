256 //for consistency with other /INTA interrupts. All interrupts works like single-shot counter
257 //Use refreshINTA() to re-enable interrupt.
258 void R8025::enableINTA_Interrupts(uint8_t periodicity)
259 {
260  intType=PERIODIC_INTERRUPT; //Record interrupt type and periodicity for use of refreshINTA() method
261  intPeriodicity=periodicity;
262 
263  uint8_t ctReg1= periodicity + 4;
264  /*
265  The above statement maps the periodicity defines from 0-3 into actual register value to be set for that mode.
266  This arithmetic operation is simpler and faster than switch-case or if-else... Hence,
267 
268  EverySecond maps to RX8025_CTRL1_CT_PER_SEC
269  EveryMinute maps to RX8025_CTRL1_CT_PER_MIN
270  EveryHour maps to RX8025_CTRL1_CT_PER_HOUR
271  EveryMonth maps to RX8025_CTRL1_CT_PER_MONTH
272 
273 */
274  //Maintain the clock in 24 Hour format
275  ctReg1 |= RX8025_CTRL1_1224_BIT ;
276  Wire.beginTransmission(RX8025_ADDRESS);
277  Wire.write(RX8025_CTRL1_REG);
278  Wire.write(ctReg1);
279  Wire.endTransmission();
280  delay(10);
281 
282  //set XST and VDSL flags.Clear all other flags
283  Wire.beginTransmission(RX8025_ADDRESS);
284  uint8_t ctReg2 = RX8025_CTRL2_XST_BIT | RX8025_CTRL2_VDSL_BIT;
285  Wire.write(RX8025_CTRL2_REG);
286  Wire.write(ctReg2);
287  Wire.endTransmission();
288  delay(10);
289 }

434 void DS3231::enableInterrupts(uint8_t periodicity)
435 {
436 
437  unsigned char ctReg=0;
438  ctReg |= 0b00011101;
439  writeRegister(DS3231_CONTROL_REG, ctReg); //CONTROL Register Address
440 
441  switch(periodicity)
442  {
443  case EverySecond:
444  writeRegister(DS3231_AL1SEC_REG, 0b10000000 ); //set AM1
445  writeRegister(DS3231_AL1MIN_REG, 0b10000000 ); //set AM2
446  writeRegister(DS3231_AL1HOUR_REG, 0b10000000 ); //set AM3
447  writeRegister(DS3231_AL1WDAY_REG, 0b10000000 ); //set AM4
448 
449  break;
450 
451  case EveryMinute:
452  writeRegister(DS3231_AL1SEC_REG, 0b00000000 ); //Clr AM1
453  writeRegister(DS3231_AL1MIN_REG, 0b10000000 ); //set AM2
454  writeRegister(DS3231_AL1HOUR_REG, 0b10000000 ); //set AM3
455  writeRegister(DS3231_AL1WDAY_REG, 0b10000000 ); //set AM4
456 
457  break;
458 
459  case EveryHour:
460  writeRegister(DS3231_AL1SEC_REG, 0b00000000 ); //Clr AM1
461  writeRegister(DS3231_AL1MIN_REG, 0b00000000 ); //Clr AM2
462  writeRegister(DS3231_AL1HOUR_REG, 0b10000000 ); //Set AM3
463  writeRegister(DS3231_AL1WDAY_REG, 0b10000000 ); //set AM4
464 
465  break;
466  }
467 }

516 //Read the temperature value from the register and convert it into float (deg C)
517 float DS3231::getTemperature()
518 {
519  int temperatureCelsius;
520  float fTemperatureCelsius;
521  uint8_t tUBYTE = readRegister(DS3231_TMP_UP_REG); //Two's complement form
522  uint8_t tLRBYTE = readRegister(DS3231_TMP_LOW_REG); //Fractional part
523 
524  if(tUBYTE & 0b10000000) //check if -ve number
525  {
526  tUBYTE ^= 0b11111111;
527  tUBYTE += 0x1;
528  fTemperatureCelsius = tUBYTE + ((tLRBYTE >> 6) * 0.25);
529  fTemperatureCelsius = fTemperatureCelsius * -1;
530  }
531  else
532  {
533  fTemperatureCelsius = tUBYTE + ((tLRBYTE >> 6) * 0.25);
534  }
535 
536  return (fTemperatureCelsius);
537 
538 }
