123 uint32_t TSL2561::calculateLux(uint16_t ch0, uint16_t ch1)
124 {
125  unsigned long chScale;
126  unsigned long channel1;
127  unsigned long channel0;
128 
129  switch (_integration)
130  {
131  case TSL2561_INTEGRATIONTIME_13MS:
132  chScale = TSL2561_LUX_CHSCALE_TINT0;
133  break;
134  case TSL2561_INTEGRATIONTIME_101MS:
135  chScale = TSL2561_LUX_CHSCALE_TINT1;
136  break;
137  default: // No scaling ... integration time = 402ms
138  chScale = (1 << TSL2561_LUX_CHSCALE);
139  break;
140  }
141 
142  // Scale for gain (1x or 16x)
143  if (!_gain) chScale = chScale << 4;
144 
145  // scale the channel values
146  channel0 = (ch0 * chScale) >> TSL2561_LUX_CHSCALE;
147  channel1 = (ch1 * chScale) >> TSL2561_LUX_CHSCALE;
148 
149  // find the ratio of the channel values (Channel1/Channel0)
150  unsigned long ratio1 = 0;
151  if (channel0 != 0) ratio1 = (channel1 << (TSL2561_LUX_RATIOSCALE+1)) / channel0;
152 
153  // round the ratio value
154  unsigned long ratio = (ratio1 + 1) >> 1;
155 
156  unsigned int b, m;
157 
158 #ifdef TSL2561_PACKAGE_CS
159  if ((ratio >= 0) && (ratio <= TSL2561_LUX_K1C))
160  {b=TSL2561_LUX_B1C; m=TSL2561_LUX_M1C;}
161  else if (ratio <= TSL2561_LUX_K2C)
162  {b=TSL2561_LUX_B2C; m=TSL2561_LUX_M2C;}
163  else if (ratio <= TSL2561_LUX_K3C)
164  {b=TSL2561_LUX_B3C; m=TSL2561_LUX_M3C;}
165  else if (ratio <= TSL2561_LUX_K4C)
166  {b=TSL2561_LUX_B4C; m=TSL2561_LUX_M4C;}
167  else if (ratio <= TSL2561_LUX_K5C)
168  {b=TSL2561_LUX_B5C; m=TSL2561_LUX_M5C;}
169  else if (ratio <= TSL2561_LUX_K6C)
170  {b=TSL2561_LUX_B6C; m=TSL2561_LUX_M6C;}
171  else if (ratio <= TSL2561_LUX_K7C)
172  {b=TSL2561_LUX_B7C; m=TSL2561_LUX_M7C;}
173  else if (ratio > TSL2561_LUX_K8C)
174  {b=TSL2561_LUX_B8C; m=TSL2561_LUX_M8C;}
175 #else
176  if ((ratio >= 0) && (ratio <= TSL2561_LUX_K1T))
177  {b=TSL2561_LUX_B1T; m=TSL2561_LUX_M1T;}
178  else if (ratio <= TSL2561_LUX_K2T)
179  {b=TSL2561_LUX_B2T; m=TSL2561_LUX_M2T;}
180  else if (ratio <= TSL2561_LUX_K3T)
181  {b=TSL2561_LUX_B3T; m=TSL2561_LUX_M3T;}
182  else if (ratio <= TSL2561_LUX_K4T)
183  {b=TSL2561_LUX_B4T; m=TSL2561_LUX_M4T;}
184  else if (ratio <= TSL2561_LUX_K5T)
185  {b=TSL2561_LUX_B5T; m=TSL2561_LUX_M5T;}
186  else if (ratio <= TSL2561_LUX_K6T)
187  {b=TSL2561_LUX_B6T; m=TSL2561_LUX_M6T;}
188  else if (ratio <= TSL2561_LUX_K7T)
189  {b=TSL2561_LUX_B7T; m=TSL2561_LUX_M7T;}
190  else if (ratio > TSL2561_LUX_K8T)
191  {b=TSL2561_LUX_B8T; m=TSL2561_LUX_M8T;}
192 #endif
193 
194  unsigned long temp;
195  temp = ((channel0 * b) - (channel1 * m));
196 
197  // do not allow negative lux value
198  if (temp < 0) temp = 0;
199 
200  // round lsb (2^(LUX_SCALE-1))
201  temp += (1 << (TSL2561_LUX_LUXSCALE-1));
202 
203  // strip off fractional portion
204  uint32_t lux = temp >> TSL2561_LUX_LUXSCALE;
205 
206  // Signal I2C had no errors
207  return lux;
208 }

210 uint32_t TSL2561::getFullLuminosity (void)
211 {
212  if (!_initialized) begin();
213 
214  // Enable the device by setting the control bit to 0x03
215  enable();
216 
217  // Wait x ms for ADC to complete
218  switch (_integration)
219  {
220  case TSL2561_INTEGRATIONTIME_13MS:
221  delay(14);
222  break;
223  case TSL2561_INTEGRATIONTIME_101MS:
224  delay(102);
225  break;
226  default:
227  delay(400);
228  break;
229  }
230 
231  uint32_t x;
232  x = read16(TSL2561_COMMAND_BIT | TSL2561_WORD_BIT | TSL2561_REGISTER_CHAN1_LOW);
233  x <<= 16;
234  x |= read16(TSL2561_COMMAND_BIT | TSL2561_WORD_BIT | TSL2561_REGISTER_CHAN0_LOW);
235 
236  disable();
237 
238  return x;
239 }
240 uint16_t TSL2561::getLuminosity (uint8_t channel) {
241 
242  uint32_t x = getFullLuminosity();
243 
244  if (channel == 0) {
245  // Reads two byte value from channel 0 (visible + infrared)
246  return (x & 0xFFFF);
247  } else if (channel == 1) {
248  // Reads two byte value from channel 1 (infrared)
249  return (x >> 16);
250  } else if (channel == 2) {
251  // Reads all and subtracts out just the visible!
252  return ( (x & 0xFFFF) - (x >> 16));
