137 void loop(void)
138 {
139  //
140  // Ping out role. Repeatedly send the current time
141  //
142 
143  if (role == role_ping_out)
144  {
145  // First, stop listening so we can talk.
146  radio.stopListening();
147 
148  // Take the time, and send it. This will block until complete
149  unsigned long time = __millis();
150  printf("Now sending %lu...",time);
151  bool ok = radio.write( &time, sizeof(unsigned long) );
152 
153  if (ok)
154  printf("ok...");
155  else
156  printf("failed.\n\r");
157 
158  // Now, continue listening
159  radio.startListening();
160 
161  // Wait here until we get a response, or timeout (250ms)
162  unsigned long started_waiting_at = __millis();
163  bool timeout = false;
164  while ( ! radio.available() && ! timeout ) {
165  // by bcatalin » Thu Feb 14, 2013 11:26 am
166  __msleep(5); //add a small delay to let radio.available to check payload
167  if (__millis() - started_waiting_at > 200 )
168  timeout = true;
169  }
170 
171  // Describe the results
172  if ( timeout )
173  {
174  printf("Failed, response timed out.\n\r");
175  }
176  else
177  {
178  // Grab the response, compare, and send to debugging spew
179  unsigned long got_time;
180  radio.read( &got_time, sizeof(unsigned long) );
181 
182  // Spew it
183  printf("Got response %lu, round-trip delay: %lu\n\r",got_time,__millis()-got_time);
184  }
185 
186  // Try again 1s later
187 // delay(1000);
188 sleep(1);
189  }
190 
191  //
192  // Pong back role. Receive each packet, dump it out, and send it back
193  //
194 
195  if ( role == role_pong_back )
196  {
197  // if there is data ready
198  if ( radio.available() )
199  {
200  // Dump the payloads until we've gotten everything
201  unsigned long got_time;
202  bool done = false;
203  while (!done)
204  {
205  // Fetch the payload, and see if this was the last one.
206  done = radio.read( &got_time, sizeof(unsigned long) );
207 
208  // Spew it
209  printf("Got payload %lu...",got_time);
210 
211  // Delay just a little bit to let the other unit
212  // make the transition to receiver
213  delay(20);
214  }
215 
216  // First, stop listening so we can talk
217  radio.stopListening();
218 
219  // Send the final one back.
220  printf("Sent response.\n\r");
221  radio.write( &got_time, sizeof(unsigned long) );
222 
223  // Now, resume listening so we catch the next packets.
224  radio.startListening();
225  }
226  }
227 }
