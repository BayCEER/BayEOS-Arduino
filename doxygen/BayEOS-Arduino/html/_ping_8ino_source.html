4  This example sends an ICMP pings every 500 milliseconds, sends the human-readable

19 byte mac[] = {0xDE, 0xAD, 0xBE, 0xEF, 0xFE, 0xED}; // max address for ethernet shield

23 SOCKET pingSocket = 0;

25 char buffer [256];
26 ICMPPing ping(pingSocket, (uint16_t)random(0, 255));

28 void setup()
29 {
30  // start Ethernet
31  Ethernet.begin(mac, ip);
32  Serial.begin(9600);
33 }

35 void loop()
36 {
37  ICMPEchoReply echoReply = ping(pingAddr, 4);
38  if (echoReply.status == SUCCESS)
39  {
40  sprintf(buffer,
41  "Reply[%d] from: %d.%d.%d.%d: bytes=%d time=%ldms TTL=%d",
42  echoReply.data.seq,
43  echoReply.addr[0],
44  echoReply.addr[1],
45  echoReply.addr[2],
46  echoReply.addr[3],
47  REQ_DATASIZE,
48  millis() - echoReply.data.time,
49  echoReply.ttl);
50  }
51  else
52  {
53  sprintf(buffer, "Echo request failed; %d", echoReply.status);
54  }
55  Serial.println(buffer);
56  delay(500);
57 }
