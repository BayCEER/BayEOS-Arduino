13 inline uint16_t _makeUint16(const uint8_t& highOrder, const uint8_t& lowOrder)
14 {
15  // make a 16-bit unsigned integer given the low order and high order bytes.

21 uint16_t _checksum(const ICMPEcho& echo)
22 {
23  // calculate the checksum of an ICMPEcho with all fields but icmpHeader.checksum populated
24  unsigned long sum = 0;
25 
26  // add the header, bytes reversed since we're using little-endian arithmetic.
27  sum += _makeUint16(echo.icmpHeader.type, echo.icmpHeader.code);
28 
29  // add id and sequence
30  sum += echo.id + echo.seq;
31 
32  // add time, one half at a time.
33  uint16_t const * time = (uint16_t const *)&echo.time;
34  sum += *time + *(time + 1);
35 
36  // add the payload
37  for (uint8_t const * b = echo.payload; b < echo.payload + sizeof(echo.payload); b+=2)
38  {
39  sum += _makeUint16(*b, *(b + 1));
40  }
41 
42  // ones complement of ones complement sum
43  sum = (sum >> 16) + (sum & 0xffff);
44  sum += (sum >> 16);
45  return ~sum;
46 }

135 Status ICMPPing::sendEchoRequest(const IPAddress& addr, const ICMPEcho& echoReq)
136 {
137  // I wish there were a better way of doing this, but if we use the uint32_t
138  // cast operator, we're forced to (1) cast away the constness, and (2) deal
139  // with an endianness nightmare.
140  uint8_t addri [] = {addr[0], addr[1], addr[2], addr[3]};
141  W5100.writeSnDIPR(_socket, addri);
142  W5100.writeSnTTL(_socket, 42);
143  // The port isn't used, becuause ICMP is a network-layer protocol. So we
144  // write zero. This probably isn't actually necessary.
145  W5100.writeSnDPORT(_socket, 0);
146 
147  uint8_t serialized [sizeof(ICMPEcho)];
148  echoReq.serialize(serialized);
149 
150  W5100.send_data_processing(_socket, serialized, sizeof(ICMPEcho));
151  W5100.execCmdSn(_socket, Sock_SEND);
152  while ((W5100.readSnIR(_socket) & SnIR::SEND_OK) != SnIR::SEND_OK)
153  {
154  if (W5100.readSnIR(_socket) & SnIR::TIMEOUT)
155  {
156  W5100.writeSnIR(_socket, (SnIR::SEND_OK | SnIR::TIMEOUT));
157  return SEND_TIMEOUT;
158  }
159  }
160  W5100.writeSnIR(_socket, SnIR::SEND_OK);
161  return SUCCESS;
162 }

164 void ICMPPing::receiveEchoReply(const ICMPEcho& echoReq, ICMPEchoReply& echoReply)
165 {
166  time_t start = millis();
167  while (millis() - start < PING_TIMEOUT)
168  {
169  if (W5100.getRXReceivedSize(_socket))
170  {
171  uint8_t ipHeader [6];
172  uint8_t buffer = W5100.readSnRX_RD(_socket);
173  W5100.read_data(_socket, (uint8_t *)buffer, ipHeader, sizeof(ipHeader));
174  buffer += sizeof(ipHeader);
175  for (int i=0; i<4; ++i) echoReply.addr[i] = ipHeader[i];
176  uint8_t dataLen = ipHeader[4];
177  dataLen = (dataLen << 8) + ipHeader[5];
178 
179  uint8_t serialized [sizeof(ICMPEcho)];
180  if (dataLen > sizeof(ICMPEcho)) dataLen = sizeof(ICMPEcho);
181  W5100.read_data(_socket, (uint8_t *)buffer, serialized, dataLen);
182  echoReply.data.deserialize(serialized);
183 
184  buffer += dataLen;
185  W5100.writeSnRX_RD(_socket, buffer);
186  W5100.execCmdSn(_socket, Sock_RECV);
187 
188  // Since there aren't any ports in ICMP, we need to manually inspect the response
189  // to see if it originated from the request we sent out.
190  if (echoReply.data.icmpHeader.type == ICMP_ECHOREP &&
191  echoReply.data.id == echoReq.id &&
192  echoReply.data.seq == echoReq.seq)
193  {
194  echoReply.ttl = W5100.readSnTTL(_socket);
195  echoReply.status = SUCCESS;
196  return;
197  }
198  }
199  }
200  echoReply.status = NO_RESPONSE;
201 }
