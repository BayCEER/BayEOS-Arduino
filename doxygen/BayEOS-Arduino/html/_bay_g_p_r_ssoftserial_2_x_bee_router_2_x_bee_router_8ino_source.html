161 void loop(void) {
162  wdcount = 0; //clear watchdog count!
163 
164  if ((millis() - last_alive) > SENDING_INTERVAL || startupframe) {
165  startupframe=0;
166  last_alive = millis();
167  client.startDataFrame(BayEOS_Float32le);
168  client.addChannelValue(millis() / 1000);
169  client.addChannelValue(myBuffer.writePos());
170  client.addChannelValue(myBuffer.readPos());
171  client.addChannelValue(tx_error);
172  client.addChannelValue(rx_ok);
173  client.addChannelValue(rx_error);
174  client.addChannelValue(client.now().get());
175  client.addChannelValue(client.getRSSI());
176  client.addChannelValue((float)analogRead(BAT_POWER_PIN)/1024*3.3*10);
177  client.writeToBuffer();
178  }
179 
180  if ((millis() - last_send) > SENDING_INTERVAL || myBuffer.available()>5000 || startupsend) {
181  //XBEE sleep mode
182  digitalWrite(RX_SLEEP_PIN,HIGH);
183  last_send = millis();
184 
185 #if SERIAL_DEBUG
186  SerialprintP("Sending via GPRS... ");
187 #endif
188  if ( (tx_res=client.sendMultiFromBuffer()) ){
189 #if SERIAL_DEBUG
190  SerialprintlnP("failed");
191 #endif
192  tx_error++;
193  rep_tx_error++;
194  client.startFrame(BayEOS_ErrorMessage);
195  client.addToPayload("TX:");
196  client.addToPayload((uint8_t)(tx_res+'0'));
197  client.writeToBuffer();
198 
199  } else {
200  rep_tx_error=0;
201  startupsend=0;
202 #if SERIAL_DEBUG
203  SerialprintlnP("OK");
204 #endif
205  }
206  if(rep_tx_error%5==4){
207  client.softSwitch();
208 
209  }
210  //XBEE sleep mode off
211  digitalWrite(RX_SLEEP_PIN,LOW);
212 
213  }
214 
215  handle_RX_data();
216 }

218 void handle_RX_data(void){
219  while(RX_SERIAL.available()){
220  xbee_rx.readPacket();
221 
222  if (xbee_rx.getResponse().isAvailable()) {
223  // got something
224  if (xbee_rx.getResponse().getApiId() == RX_16_RESPONSE) {
225  // got a rx16 packet
226  xbee_rx.getResponse().getRx16Response(rx16);
227  if (xbee_rx.getResponse().isError()) {
228  rx_error++;
229  // get the error code
230  } else {
231  rx_ok++;
232 #if SERIAL_DEBUG
233  SerialprintlnP("Got RX-Packet");
234 #endif
235  last_rx_rssi=rx16.getRssi();
236  client.startRoutedFrame(rx16.getRemoteAddress16(), rx_panid,
237  last_rx_rssi);
238  for (uint8_t i = 0; i < rx16.getDataLength(); i++) {
239  client.addToPayload(rx16.getData(i));
240  }
241  }
242  client.writeToBuffer();
243  }
244  }
245  }
