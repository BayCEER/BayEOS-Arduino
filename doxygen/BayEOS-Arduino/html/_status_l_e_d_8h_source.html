1 /*
2  * StatusLED Class
3  * Class is designed to be used with TimerOne Class
4  *
5  * use
6  * led.init(); //initialize pins
7  * Timer1.initialize(80000); //initialize timer to 80ms
8  * Timer1.attachInterrupt(led.callback); //attach led callback
9  * in setup()
10  *
11  * to blink use the macros
12  * BLINK_RED(times)
13  * BLINK_GREEN(times)
14  *
15  * Alternative there are the methods blink_red, blink_green
16  * using delay()
17  *
18  * STATUS_LED_PIN defaults to 11
19  * ERROR_LED_PIN defaults to 12
20  *
21  * to speed up ISR direct PORT Manipulation is used
22  * works only with pin layout of ATMEGA328 and

113 static void callback(void){
114  if(status_blink){
115  status_blink--;
116 #if PORT_MANIPULATION!=1
117  if((status_blink & 0x3)>0x1)
118  digitalWrite(STATUS_LED_PIN,(status_blink & 0x3)==0x3);
119 #endif
120  }
121  if(error_blink){
122  error_blink--;
123 #if PORT_MANIPULATION!=1

126 #endif
127  }
128 #if PORT_MANIPULATION==1
129  STATUS_LED_PORT=STATUS_LED_PORT&PIN_MASK |
130  ((error_blink & 0x3)==0x3)<<ERROR_LED_BIT |
131  ((status_blink & 0x3)==0x3)<<STATUS_LED_BIT;
