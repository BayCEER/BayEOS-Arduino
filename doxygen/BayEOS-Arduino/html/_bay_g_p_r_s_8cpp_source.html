655 uint8_t BayGPRSInterface::wait_forPGM(const char* str,uint16_t timeout,uint8_t bytes, char* buffer){
656  uint8_t length=0;
657  while (true) {
658  _pgm_buffer[length]=pgm_read_byte(str);
659  if (! _pgm_buffer[length]) break;
660  str++;
661  length++;
662  }
663 
664  if(wait_for_available(&timeout,length))
665  return 2;
666  uint8_t offset=0;
667  char c;
668  while(true){
669  if(wait_for_available(&timeout))
670  return 2;
671  c=read();
672 #if BayGPRS_DEBUG_OUTPUT
673  Serial.print(c);
674 #endif
675 
676  if(offset<length){
677  if(c==_pgm_buffer[offset])
678  offset++;
679  else if(offset>0){
680  skipChars();
681  return 1;
682  }
683  }
684 
685  if(offset==length){
686  if(bytes && buffer!=NULL){
687  if(wait_for_available(&timeout,bytes))
688  return 2;
689  offset=0;
690  while(offset<bytes){
691  buffer[offset]=read();
692  offset++;
693  }
694  buffer[offset]=0;
695  }
696  skipChars();
697  return 0;
698  }
